#include <WiFi.h>
#include <HTTPClient.h>
#include <ArduinoJson.h>
#include <DHT.h>
#include <ESP32Servo.h>

// ========== Wi-Fi НАСТРОЙКИ (замените на свои) ==========
const char* ssid = "Badhome";
const char* password = "adminadmin";

// ========== СЕРВЕР (HTTP, локальная сеть — используйте WiFiClient, не WiFiClientSecure) ==========
const char* serverUrl = "http://10.190.201.71:8000/api/update";

// ========== ПИНЫ ==========
#define PIR_LIVING     13
#define DHT_LIVING     14
#define MQ2_LIVING     35
#define LED_DAY_LIVING 26
#define LED_NIGHT_LIVING 27
#define FAN_LIVING     32

#define PIR_BEDROOM     15
#define DHT_BEDROOM     4
#define MQ2_BEDROOM     19
#define LED_DAY_BEDROOM 16
#define LED_NIGHT_BEDROOM 17
#define FAN_BEDROOM     18

#define SMART_PLUG_LIVING_1  23   // socket_tv
#define SMART_PLUG_LIVING_2  25   // socket_ac
#define SMART_PLUG_BEDROOM_1 12   // socket_tv
#define SMART_PLUG_BEDROOM_2 33   // socket_ac
#define LIGHT_STRIP         2     // rgb
#define LIGHT_SENSOR        34
#define SERVO_PIN           5

#define DHT_TYPE DHT11

// ========== ТАЙМИНГИ ==========
const unsigned long SENSOR_READ_INTERVAL = 2000;   // опрос датчиков
const unsigned long POST_INTERVAL = 2000;        // отправка датчиков на сервер
const unsigned long GET_INTERVAL = 2000;          // запрос состояния устройств

// ========== НАСТРОЙКИ FS5103R ==========
const unsigned long BLIND_MOVE_TIME = 2000;   // время полного хода (мс)
const int BLIND_STOP = 92;
const int BLIND_UP_SPEED = 180;
const int BLIND_DOWN_SPEED = 0;

// ========== ГЛОБАЛЬНЫЕ ПЕРЕМЕННЫЕ ==========
DHT dhtLiving(DHT_LIVING, DHT_TYPE);
DHT dhtBedroom(DHT_BEDROOM, DHT_TYPE);
Servo blindServo;

struct Room {
  bool motion;
  float temperature;
  float humidity;
  bool gasDetected;
  unsigned long lastReadTime;
  unsigned long motionDetectedTime;
  bool motionWasDetected;
};

Room living, bedroom;

enum BlindState { BLIND_IDLE, BLIND_MOVING_UP, BLIND_MOVING_DOWN };
BlindState blindState = BLIND_IDLE;
unsigned long blindMoveStartTime = 0;
bool targetBlindsUp = true;
bool lastAppliedBlindsUp = true;

// Состояния устройств, полученные от сервера
struct DeviceState {
  bool day_light_1 = false;
  bool night_light_1 = false;
  bool vent_1 = false;
  bool socket_tv_1 = false;
  bool socket_ac_1 = false;
  bool rgb_1 = false;

  bool day_light_2 = false;
  bool night_light_2 = false;
  bool vent_2 = false;
  bool socket_tv_2 = false;
  bool socket_ac_2 = false;
  bool servo_2 = false;
};

DeviceState deviceState;

unsigned long lastSensorRead = 0;
unsigned long lastPostTime = 0;
unsigned long lastGetTime = 0;

// ========== ФУНКЦИИ УПРАВЛЕНИЯ УСТРОЙСТВАМИ ==========
void setLightStrip(bool on) {
  digitalWrite(LIGHT_STRIP, on ? LOW : HIGH);
}

void startBlindMove(bool up) {
  if (blindState != BLIND_IDLE) {
    blindServo.write(BLIND_STOP);
    delay(20);
  }
  if (up) {
    blindServo.write(BLIND_UP_SPEED);
    blindState = BLIND_MOVING_UP;
  } else {
    blindServo.write(BLIND_DOWN_SPEED);
    blindState = BLIND_MOVING_DOWN;
  }
  blindMoveStartTime = millis();
}

void stopBlindMove() {
  if (blindState != BLIND_IDLE) {
    blindServo.write(BLIND_STOP);
    blindState = BLIND_IDLE;
  }
}

void setBlindsTarget(bool up) {
  if (targetBlindsUp == up && blindState == BLIND_IDLE) return;
  targetBlindsUp = up;
  if (blindState == BLIND_IDLE) {
    startBlindMove(up);
  }
}

void updateBlindsMovement() {
  if (blindState != BLIND_IDLE && millis() - blindMoveStartTime >= BLIND_MOVE_TIME) {
    stopBlindMove();
  }
}

void setSmartPlug(int room, const String& type, bool state) {
  int pin = -1;
  if (room == 1) {
    if (type == "socket_tv") pin = SMART_PLUG_LIVING_1;
    else if (type == "socket_ac") pin = SMART_PLUG_LIVING_2;
  } else if (room == 2) {
    if (type == "socket_tv") pin = SMART_PLUG_BEDROOM_1;
    else if (type == "socket_ac") pin = SMART_PLUG_BEDROOM_2;
  }
  if (pin != -1) {
    digitalWrite(pin, state ? LOW : HIGH);
  }
}

// Применить все состояния из deviceState к выходам
void applyDeviceStates() {
  // Комната 1
  digitalWrite(LED_DAY_LIVING, deviceState.day_light_1 ? HIGH : LOW);
  digitalWrite(LED_NIGHT_LIVING, deviceState.night_light_1 ? HIGH : LOW);
  digitalWrite(FAN_LIVING, deviceState.vent_1 ? LOW : HIGH);   // вентилятор активный LOW
  setSmartPlug(1, "socket_tv", deviceState.socket_tv_1);
  setSmartPlug(1, "socket_ac", deviceState.socket_ac_1);
  setLightStrip(deviceState.rgb_1);   // RGB-лента

  // Комната 2
  digitalWrite(LED_DAY_BEDROOM, deviceState.day_light_2 ? HIGH : LOW);
  digitalWrite(LED_NIGHT_BEDROOM, deviceState.night_light_2 ? HIGH : LOW);
  digitalWrite(FAN_BEDROOM, deviceState.vent_2 ? LOW : HIGH);
  setSmartPlug(2, "socket_tv", deviceState.socket_tv_2);
  setSmartPlug(2, "socket_ac", deviceState.socket_ac_2);
  setBlindsTarget(deviceState.servo_2);   // управление жалюзи
}

// ========== СБОР ДАННЫХ С ДАТЧИКОВ ==========
void readSensors() {
  unsigned long now = millis();
  if (now - lastSensorRead >= SENSOR_READ_INTERVAL) {
    lastSensorRead = now;

    // Гостиная
    living.motion = digitalRead(PIR_LIVING);
    float t = dhtLiving.readTemperature();
    float h = dhtLiving.readHumidity();
    if (!isnan(t)) living.temperature = t;
    if (!isnan(h)) living.humidity = h;
    living.gasDetected = (digitalRead(MQ2_LIVING) == LOW);

    // Спальня
    bedroom.motion = digitalRead(PIR_BEDROOM);
    t = dhtBedroom.readTemperature();
    h = dhtBedroom.readHumidity();
    if (!isnan(t)) bedroom.temperature = t;
    if (!isnan(h)) bedroom.humidity = h;
    bedroom.gasDetected = (digitalRead(MQ2_BEDROOM) == LOW);
  }
}

// ========== ОТПРАВКА ДАННЫХ НА СЕРВЕР (POST) ==========
void sendSensorData() {
  if (WiFi.status() != WL_CONNECTED) return;

  StaticJsonDocument<1024> doc;
  JsonArray readings = doc.createNestedArray("readings");

  // Гостиная (room_id = 1)
  JsonObject obj1 = readings.createNestedObject();
  obj1["sensor_name"] = "temperature";
  obj1["room_id"] = 1;
  obj1["value"] = living.temperature;

  JsonObject obj2 = readings.createNestedObject();
  obj2["sensor_name"] = "humidity";
  obj2["room_id"] = 1;
  obj2["value"] = living.humidity;

  JsonObject obj3 = readings.createNestedObject();
  obj3["sensor_name"] = "light_level";
  obj3["room_id"] = 1;
  obj3["value"] = analogRead(LIGHT_SENSOR);

  JsonObject obj4 = readings.createNestedObject();
  obj4["sensor_name"] = "movement";
  obj4["room_id"] = 1;
  obj4["value"] = living.motion ? 1.0f : 0.0f;

  JsonObject obj5 = readings.createNestedObject();
  obj5["sensor_name"] = "gas";
  obj5["room_id"] = 1;
  obj5["value"] = living.gasDetected ? 1.0f : 0.0f;

  // Спальня (room_id = 2)
  JsonObject obj6 = readings.createNestedObject();
  obj6["sensor_name"] = "temperature";
  obj6["room_id"] = 2;
  obj6["value"] = bedroom.temperature;

  JsonObject obj7 = readings.createNestedObject();
  obj7["sensor_name"] = "humidity";
  obj7["room_id"] = 2;
  obj7["value"] = bedroom.humidity;

  JsonObject obj8 = readings.createNestedObject();
  obj8["sensor_name"] = "movement";
  obj8["room_id"] = 2;
  obj8["value"] = bedroom.motion ? 1.0f : 0.0f;

  JsonObject obj9 = readings.createNestedObject();
  obj9["sensor_name"] = "gas";
  obj9["room_id"] = 2;
  obj9["value"] = bedroom.gasDetected ? 1.0f : 0.0f;

  String jsonStr;
  serializeJson(doc, jsonStr);

  WiFiClient client;
  HTTPClient http;
  http.begin(client, serverUrl);
  http.addHeader("Content-Type", "application/json");
  int httpCode = http.POST(jsonStr);

  if (httpCode > 0) {
    Serial.printf("POST ответ: %d\n", httpCode);
    String payload = http.getString();
    Serial.println(payload);
  } else {
    Serial.printf("POST ошибка: %s\n", http.errorToString(httpCode).c_str());
  }
  http.end();
}

// ========== ПОЛУЧЕНИЕ СОСТОЯНИЙ УСТРОЙСТВ (GET) ==========
void fetchDeviceStates() {
  if (WiFi.status() != WL_CONNECTED) return;

  WiFiClient client;
  HTTPClient http;
  http.begin(client, serverUrl);
  int httpCode = http.GET();

  if (httpCode == HTTP_CODE_OK) {
    String payload = http.getString();
    DynamicJsonDocument doc(2048);
    deserializeJson(doc, payload);

    // Сценарий и auto_mode не используются для управления выходами,
    // но можно вывести в Serial для отладки
    const char* scenario = doc["scenario"];
    bool autoMode = doc["auto_mode"];
    Serial.printf("GET: scenario=%s, auto_mode=%d\n", scenario, autoMode);

    // Очищаем предыдущие состояния
    DeviceState newState;

    // Парсим массив devices
    JsonArray devices = doc["devices"];
    for (JsonObject dev : devices) {
      const char* type = dev["device_type"];
      int room = dev["room_id"];
      bool isOn = dev["is_on"];

      if (room == 1) {
        if (strcmp(type, "day_light") == 0) newState.day_light_1 = isOn;
        else if (strcmp(type, "night_light") == 0) newState.night_light_1 = isOn;
        else if (strcmp(type, "vent") == 0) newState.vent_1 = isOn;
        else if (strcmp(type, "socket_tv") == 0) newState.socket_tv_1 = isOn;
        else if (strcmp(type, "socket_ac") == 0) newState.socket_ac_1 = isOn;
        else if (strcmp(type, "rgb") == 0) newState.rgb_1 = isOn;
      } else if (room == 2) {
        if (strcmp(type, "day_light") == 0) newState.day_light_2 = isOn;
        else if (strcmp(type, "night_light") == 0) newState.night_light_2 = isOn;
        else if (strcmp(type, "vent") == 0) newState.vent_2 = isOn;
        else if (strcmp(type, "socket_tv") == 0) newState.socket_tv_2 = isOn;
        else if (strcmp(type, "socket_ac") == 0) newState.socket_ac_2 = isOn;
        else if (strcmp(type, "servo") == 0) newState.servo_2 = isOn;
      }
    }

    // Применяем новое состояние, если оно изменилось
    if (memcmp(&deviceState, &newState, sizeof(DeviceState)) != 0) {
      deviceState = newState;
      applyDeviceStates();
    }
  } else {
    Serial.printf("GET ошибка: %s\n", http.errorToString(httpCode).c_str());
  }
  http.end();
}

// ========== ПОДКЛЮЧЕНИЕ К WI-FI ==========
void setupWiFi() {
  Serial.print("Подключение к Wi-Fi");
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nWi-Fi подключён");
  Serial.print("IP адрес: ");
  Serial.println(WiFi.localIP());
}

// ========== НАЧАЛЬНАЯ НАСТРОЙКА ==========
void setup() {
  Serial.begin(115200);
  delay(1000);
  Serial.println("Умный дом с серверным управлением");

  // Настройка пинов
  pinMode(PIR_LIVING, INPUT); pinMode(PIR_BEDROOM, INPUT);
  pinMode(MQ2_LIVING, INPUT_PULLUP); pinMode(MQ2_BEDROOM, INPUT_PULLUP);
  pinMode(LIGHT_SENSOR, INPUT);
  pinMode(LED_DAY_LIVING, OUTPUT); pinMode(LED_NIGHT_LIVING, OUTPUT);
  pinMode(LED_DAY_BEDROOM, OUTPUT); pinMode(LED_NIGHT_BEDROOM, OUTPUT);
  pinMode(FAN_LIVING, OUTPUT); pinMode(FAN_BEDROOM, OUTPUT);
  pinMode(SMART_PLUG_LIVING_1, OUTPUT); pinMode(SMART_PLUG_LIVING_2, OUTPUT);
  pinMode(SMART_PLUG_BEDROOM_1, OUTPUT); pinMode(SMART_PLUG_BEDROOM_2, OUTPUT);
  pinMode(LIGHT_STRIP, OUTPUT);

  // Инициализация выходов (все выключены)
  digitalWrite(LED_DAY_LIVING, LOW); digitalWrite(LED_NIGHT_LIVING, LOW);
  digitalWrite(LED_DAY_BEDROOM, LOW); digitalWrite(LED_NIGHT_BEDROOM, LOW);
  digitalWrite(FAN_LIVING, HIGH); digitalWrite(FAN_BEDROOM, HIGH);
  digitalWrite(SMART_PLUG_LIVING_1, HIGH); digitalWrite(SMART_PLUG_LIVING_2, HIGH);
  digitalWrite(SMART_PLUG_BEDROOM_1, HIGH); digitalWrite(SMART_PLUG_BEDROOM_2, HIGH);
  digitalWrite(LIGHT_STRIP, HIGH);

  dhtLiving.begin(); dhtBedroom.begin();

  // Сервопривод
  blindServo.attach(SERVO_PIN);
  blindServo.write(BLIND_STOP);

  // Подключение к Wi-Fi
  setupWiFi();

  // Первичный опрос состояния устройств
  fetchDeviceStates();

  // Установка начальных таймеров
  lastSensorRead = millis();
  lastPostTime = millis();
  lastGetTime = millis();
}

// ========== ГЛАВНЫЙ ЦИКЛ ==========
void loop() {
  unsigned long now = millis();

  // Чтение датчиков (не блокирующее)
  readSensors();

  // Отправка данных на сервер (POST) с интервалом POST_INTERVAL
  if (now - lastPostTime >= POST_INTERVAL) {
    lastPostTime = now;
    sendSensorData();
  }

  // Получение состояния устройств (GET) с интервалом GET_INTERVAL
  if (now - lastGetTime >= GET_INTERVAL) {
    lastGetTime = now;
    fetchDeviceStates();
  }

  // Обновление движения жалюзи (неблокирующее)
  updateBlindsMovement();

  // Небольшая задержка для стабильности цикла
  delay(1);
}
